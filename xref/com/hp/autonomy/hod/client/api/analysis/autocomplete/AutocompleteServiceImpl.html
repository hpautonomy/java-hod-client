18  import java.util.List;

20  /**
21   * Default implementation of an AutocompleteService.
22   */
23  public class AutocompleteServiceImpl implements AutocompleteService {
24      private final AutocompleteBackend backend;
25      private final Requester<?, TokenType.Simple> requester;
26  
27      public AutocompleteServiceImpl(final HodServiceConfig<?, TokenType.Simple> config) {
28          requester = config.getRequester();
29          backend = config.getRestAdapter().create(AutocompleteBackend.class);
30      }
31  
32      @Override
33      public List<String> getSuggestions(final String text) throws HodErrorException {
34          return requester.makeRequest(AutocompleteResponse.class, new BackendCaller<>(text)).getWords();
35      }
36  
37      @Override
38      public List<String> getSuggestions(final String text, final TokenProxy<?, TokenType.Simple> tokenProxy) throws HodErrorException {
39          return requester.makeRequest(tokenProxy, AutocompleteResponse.class, new BackendCaller<>(text)).getWords();
40      }
41  
42      @Data
43      private class BackendCaller<E extends EntityType> implements Requester.BackendCaller<E, TokenType.Simple> {
44          private final String text;
45  
46          @Override
47          public Response makeRequest(final AuthenticationToken<? extends E, ? extends TokenType.Simple> token) throws HodErrorException {
48              return backend.getSuggestions(token, text);
49          }
50      }
51  }
