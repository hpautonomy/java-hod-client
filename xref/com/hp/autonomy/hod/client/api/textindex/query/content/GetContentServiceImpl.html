22  /**
23   * Default implementation of GetContentService
24   * <p/>
25   * If a custom return type is not required, use the {@link #documentsService(HodServiceConfig)} static factory method
26   * @param <T> The desired return type of the services methods
27   */
28  @SuppressWarnings("WeakerAccess")
29  public class GetContentServiceImpl<T extends Serializable> implements GetContentService<T> {
30      
31      private final GetContentBackend getContentBackend;
32      private final Requester<?, TokenType.Simple> requester;
33      private final JavaType returnType;
34  
35      /**
36       * Create a new GetContentServiceImpl of the desired parameterized type
37       * @param config The configuration to use
38       * @param documentType The desired document type for service methods. This type must have the correct Jackson annotations
39       * to read HP Haven OnDemand responses
40       */
41      public GetContentServiceImpl(final HodServiceConfig<?, TokenType.Simple> config, final Class<T> documentType) {
42          getContentBackend = config.getRestAdapter().create(GetContentBackend.class);
43          requester = config.getRequester();
44          returnType = config.getObjectMapper().getTypeFactory().constructParametrizedType(QueryResults.class, QueryResults.class, documentType);
45      }
46  
47      /**
48       * Create a new GetContentServiceImpl of type {@link QueryResults}
49       * @param hodServiceConfig The configuration to use
50       * @return A new {@literal GetContentService<Document>}
51       */
52      public static GetContentServiceImpl<Document> documentsService(final HodServiceConfig<?, TokenType.Simple> hodServiceConfig) {
53          return new GetContentServiceImpl<>(hodServiceConfig, Document.class);
54      }
55      
56      @Override
57      public QueryResults<T> getContent(final List<String> indexReference, final ResourceIdentifier index, final GetContentRequestBuilder params) throws HodErrorException {
58          return requester.unsafeMakeRequest(returnType, getBackendCaller(indexReference, index, params));
59      }
60  
61      @Override
62      public QueryResults<T> getContent(final TokenProxy<?, TokenType.Simple> tokenProxy, final List<String> indexReference, final ResourceIdentifier index, final GetContentRequestBuilder params) throws HodErrorException {
63          return requester.unsafeMakeRequest(tokenProxy, returnType, getBackendCaller(indexReference, index, params));
64      }
65  
66      private Requester.BackendCaller<EntityType, TokenType.Simple> getBackendCaller(final List<String> indexReference, final ResourceIdentifier indexes, final GetContentRequestBuilder params) {
67          return authenticationToken -> getContentBackend.getContent(authenticationToken, indexReference, indexes, params.build());
68      }
69  }
