17  /**
18   * Service for making authentication requests to HP Haven OnDemand
19   */
20  interface AuthenticationBackend {

25      String NONCE_PARAMETER = "nonce";

34      /**
35       * Acquire a token for an application
36       * @param apiKey The API key of the application
37       * @param applicationName The name of the application
38       * @param domain The domain of the application
39       * @param tokenType The type of the resulting token
40       * @return A response containing a token for use with HP Haven OnDemand
41       * @throws HodErrorException
42       */
43      @POST("/2/authenticate/application")
44      @Multipart
45      Response authenticateApplication(
46          @Header("apiKey") ApiKey apiKey,
47          @Part(NAME_PARAMETER) String applicationName,
48          @Part(DOMAIN_PARAMETER) String domain,
49          @Part(TOKEN_TYPE_PARAMETER) String tokenType
50      ) throws HodErrorException;
51  
52      /**
53       * Acquire a token for a user
54       * @param apiKey The API key of the user
55       * @param applicationName The name of the application
56       * @param applicationDomain The domain of the application
57       * @param tokenType The type of the resulting token
58       * @return A response containing a token for use with HP Haven OnDemand
59       * @throws HodErrorException
60       */
61      @POST("/2/authenticate/user")
62      @Multipart
63      Response authenticateUser(
64          @Header("apiKey") ApiKey apiKey,
65          @Part("application_name") String applicationName,
66          @Part("application_domain") String applicationDomain,
67          @Part(TOKEN_TYPE_PARAMETER) String tokenType
68      ) throws HodErrorException;
69  
70      /**
71       * Acquire a token for a user in a given user store
72       * @param apiKey The API key of the user
73       * @param applicationName The name of the application
74       * @param applicationDomain The domain of the application
75       * @param tokenType The type of the resulting token
76       * @param userStore The name of the user store
77       * @param storeDomain The domain of the user store
78       * @return A response containing a token for use with HP Haven OnDemand
79       * @throws HodErrorException
80       */
81      @POST("/2/authenticate/user")
82      @Multipart
83      Response authenticateUser(
84          @Header("apiKey") ApiKey apiKey,
85          @Part("application_name") String applicationName,
86          @Part("application_domain") String applicationDomain,
87          @Part("token_type") String tokenType,
88          @Part("user_store_name") String userStore,
89          @Part("user_store_domain") String storeDomain
90      ) throws HodErrorException;
91  
92      @POST("/2/authenticate/developer")
93      @Multipart
94      Response authenticateDeveloper(
95          @Header("apiKey") ApiKey apiKey,
96          @Part("tenant") String tenant,
97          @Part("email") String email
98      ) throws HodErrorException;
99  
100     /**

107     @POST("/2/authenticate/unbound")

135         @Part(TOKEN_TYPE_PARAMETER) String tokenType
