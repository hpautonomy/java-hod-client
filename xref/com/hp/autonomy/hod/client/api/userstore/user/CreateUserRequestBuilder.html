8   import com.hp.autonomy.hod.client.api.developer.UserAuthMode;
9   import com.hp.autonomy.hod.client.util.MultiMap;
10  import lombok.Data;
11  import lombok.experimental.Accessors;
12  
13  import java.util.LinkedList;
14  import java.util.List;
15  import java.util.Map;
16  import java.util.stream.Collectors;
17  
18  /**
19   * Helper class for building up optional parameters for the create user API. The default value
20   * for all parameters is null. Null parameters will not be sent to HP Haven OnDemand
21   */
22  @Data
23  @Accessors(chain = true)
24  public class CreateUserRequestBuilder {
25  
26      /**
27       * @param userMessage The user_message create user parameter
28       */
29      private String userMessage;
30  
31      /**
32       * @param userMode The mode by which users must authenticate
33       */
34      private UserAuthMode userMode;
35  
36      /**
37       * @param metadata Metadata keys and values to associate with the user
38       */
39      private Map<String, ?> metadata;
40  
41      Map<String, Object> build() {
42          final Map<String, Object> map = new MultiMap<>();
43          map.put("user_message", userMessage);
44          map.put("user_mode", userMode);
45  
46          if (metadata != null) {
47              final List<Metadata<?>> metadataList = metadata.entrySet().stream().map(entry -> new Metadata<>(entry.getKey(), entry.getValue())).collect(Collectors.toCollection(LinkedList::new));
48  
49              map.put("metadata", metadataList);
50          }
51  
52          return map;
53      }
54  }
