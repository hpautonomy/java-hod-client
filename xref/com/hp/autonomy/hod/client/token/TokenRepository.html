6   package com.hp.autonomy.hod.client.token;
7   
8   import com.hp.autonomy.hod.client.api.authentication.AuthenticationToken;
9   import com.hp.autonomy.hod.client.api.authentication.EntityType;
10  import com.hp.autonomy.hod.client.api.authentication.TokenType;
11  
12  import java.io.IOException;
13  
14  /**
15   * A TokenRepository stores tokens such that they can be refreshed as instructed by HP Haven OnDemand. Note that the
16   * token retrieved by the get method may not be the same token that was originally inserted, but will have the same
17   * capabilities.
18   *
19   * Repositories may automatically remove expired tokens
20   */
21  public interface TokenRepository {
22  
23      /**
24       * Insert a token into the repository for later retrieval
25       * @param token The {@link AuthenticationToken} to store
26       * @return A key which can be used to retrieve the token
27       * @throws IOException If an IO error occurred inserting the token
28       * @throws IllegalArgumentException If the token has expired
29       */
30      <E extends EntityType, T extends TokenType> TokenProxy<E, T> insert(AuthenticationToken<E, T> token) throws IOException;
31  
32      /**
33       * Update the token in the repository associated with the given key. If the key is not in the repository, the new
34       * token will not be stored
35       * @param key The key to associate with the token
36       * @param newToken The new token for the key
37       * @return The old token associated with the key, or null if no key was associated
38       * @throws IOException If an IO error occurred updating the token
39       * @throws IllegalArgumentException If the token has expired
40       */
41      <E extends EntityType, T extends TokenType> AuthenticationToken<E, T> update(TokenProxy<E, T> key, AuthenticationToken<E, T> newToken) throws IOException;
42  
43      /**
44       * Retrieve the token associated with the given key
45       * @param key The key with which to retrieve a token
46       * @return The token associated with the key, or null if no token is associated with the key
47       * @throws IOException If an IO error occurred retrieving the token
48       */
49      <E extends EntityType, T extends TokenType> AuthenticationToken<E, T> get(TokenProxy<E, T> key) throws IOException;
50  
51      /**
52       * Removes the token associated with the given key
53       * @param key The key for which a token should be removed
54       * @return The old token associated with the key
55       * @throws IOException If an IO error occurred removing the token
56       */
57      <E extends EntityType, T extends TokenType> AuthenticationToken<E, T> remove(TokenProxy<E, T> key) throws IOException;
58  
59  }
