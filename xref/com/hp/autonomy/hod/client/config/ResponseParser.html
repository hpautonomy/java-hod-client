24  /**
25   * Parses a response from HP Haven OnDemand. If a token is refreshed, this will update the token repository.
26   */
27  class ResponseParser {
28  
29      private static final String TOKEN_HEADER_NAME = "refresh_token";
30  
31      private final TokenRepository tokenRepository;
32      private final ObjectMapper objectMapper;
33  
34      /**
35       * Create a new ResponseParser
36       * @param tokenRepository The token repository to update if required
37       * @param objectMapper The object mapper to use to convert the token and the response body
38       */
39      ResponseParser(final TokenRepository tokenRepository, final ObjectMapper objectMapper) {
40          this.tokenRepository = tokenRepository;
41          this.objectMapper = objectMapper;
42      }
43  
44      /**
45       * Parse the response into a type represented by the Class object. Checks for a refreshed token and updates the token
46       * repository accordingly.
47       * @param <T> The desired type of the response body
48       * @param tokenProxy The token proxy to update in the token repository if required
49       * @param clazz A class object representing the desired type of the response body
50       * @param response The response to parse
51       * @return An object representing the result from HP Haven OnDemand
52       */
53      <T> T parseResponse(final TokenProxy<?, ?> tokenProxy, final Class<T> clazz, final Response response) {
54          return unsafeParseResponse(tokenProxy, objectMapper.constructType(clazz), response);
55      }
56  
57      /**
58       * Parse the response into a type represented by the TypeReference object. Checks for a refreshed token and updates
59       * the token repository accordingly.
60       * @param <T> The desired type of the response body
61       * @param tokenProxy The token proxy to update in the token repository if required
62       * @param typeReference A type reference object representing the desired type of the response body
63       * @param response The response to parse
64       * @return An object representing the result from HP Haven OnDemand
65       */
66      <T> T parseResponse(final TokenProxy<?, ?> tokenProxy, final TypeReference<T> typeReference, final Response response) {
67          return unsafeParseResponse(tokenProxy, objectMapper.getTypeFactory().constructType(typeReference), response);
68      }
69  
70      /**
71       * Parse the response into a type represented by the JavaType object. Checks for a refreshed token and updates
72       * the token repository accordingly. This method does not guarantee that the type parameter T represents the given
73       * JavaType. If possible, use one of the safe parseResponse methods instead.
74       * @param <T> The desired type of the response body
75       * @param tokenProxy The token proxy to update in the token repository if required
76       * @param type A JavaType representing the desired type of the response body
77       * @param response The response to parse
78       * @return An object representing the result from HP Haven OnDemand
79       */
80      <T> T unsafeParseResponse(final TokenProxy<?, ?> tokenProxy, final JavaType type, final Response response) {
81          checkRefresh(tokenProxy, response);
82  
83          try {
84              return objectMapper.readValue(response.getBody().in(), type);
85          } catch (final IOException e) {
86              throw new RuntimeException(e);
87          }
88      }
89  
90      /**
91       * Checks the response headers for a refreshed authentication token then returns the response body as an input stream.
92       * @param tokenProxy The token proxy to update in the token repository if required
93       * @param response The response to read
94       * @return The response body as an input stream
95       */
96      InputStream parseResponse(final TokenProxy<?, ?> tokenProxy, final Response response) {
97          checkRefresh(tokenProxy, response);
98  
99          try {
100             return response.getBody().in();
101         } catch (final IOException e) {
102             throw new RuntimeException(e);
103         }
104     }
105 
106     private <E extends EntityType, T extends TokenType> void checkRefresh(final TokenProxy<E, T> tokenProxy, final Response response) {
107         final List<Header> headers = response.getHeaders();
108 
109         for(final Header header : headers) {
110             if(TOKEN_HEADER_NAME.equals(header.getName())) {
111                 try {
112                     final AuthenticationToken.Json parsedHeader = objectMapper.readValue(header.getValue(), AuthenticationToken.Json.class);
113                     final AuthenticationToken<E, T> token = parsedHeader.buildToken(tokenProxy.getEntityType(), tokenProxy.getTokenType());
114                     tokenRepository.update(tokenProxy, token);
115                 } catch (final IOException e) {
116                     throw new TokenRepositoryException(e);
117                 }
118 
119                 break;
120             }
121         }
122     }
