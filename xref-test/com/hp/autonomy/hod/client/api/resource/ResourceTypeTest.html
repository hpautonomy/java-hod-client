31          final ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
32          final ObjectOutputStream objectOutputStream = new ObjectOutputStream(byteArrayOutputStream);
33          objectOutputStream.writeObject(ResourceType.TEXT_INDEX);
34  
35          final ObjectInputStream objectInputStream = new ObjectInputStream(new ByteArrayInputStream(byteArrayOutputStream.toByteArray()));
36          final ResourceType resourceType = (ResourceType) objectInputStream.readObject();

38          assertThat(resourceType, is(ResourceType.TEXT_INDEX));
39      }

41      // this test uses reflection to check allOf contains all the types
42      // we don't use reflection in the class for performance reasons
43      @Test
44      public void testAllOf() throws IllegalAccessException {
45          final Field[] fields = ResourceType.class.getFields();
46  
47          final Set<ResourceType> resourceTypes = new HashSet<>();
48  
49          final ResourceType dummy = ResourceType.CONNECTOR;
50  
51          for(final Field field : fields) {
52              final int modifiers = field.getModifiers();
53              if (Modifier.isPublic(modifiers) && Modifier.isStatic(modifiers) && Modifier.isFinal(modifiers) && field.getType() == ResourceType.class) {
54                  resourceTypes.add((ResourceType) field.get(dummy));
55              }
56          }
57  
58          assertThat(ResourceType.allOf(), is(resourceTypes));
59      }
60  
61      @Test
62      public void testOf() {
63          final Set<ResourceType> resourceTypes = ResourceType.of(ResourceType.TEXT_INDEX, ResourceType.CONNECTOR);
64  
65          assertThat(resourceTypes, hasSize(2));
66          assertThat(resourceTypes, hasItems(ResourceType.TEXT_INDEX, ResourceType.CONNECTOR));
67      }

69      @Test
70      public void testComplementOf() {
71          final Set<ResourceType> complement = ResourceType.complementOf(ResourceType.of(ResourceType.CONNECTOR));
72  
73          assertThat(complement, hasSize(ResourceType.allOf().size() - 1));
74          assertThat(complement, not(hasItem(ResourceType.CONNECTOR)));
