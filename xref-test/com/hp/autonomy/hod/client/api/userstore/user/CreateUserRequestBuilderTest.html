20  public class CreateUserRequestBuilderTest {
21      private static final String USER_MESSAGE_KEY = "user_message";
22      private static final String METADATA_KEY = "metadata";
23  
24      @Test
25      public void buildWithNoParameters() {
26          final Map<String, Object> output = new CreateUserRequestBuilder().build();
27  
28          for (final Map.Entry<String, Object> entry : output.entrySet()) {
29              assertThat(entry.getValue(), is(nullValue()));
30          }
31      }
32  
33      @Test
34      public void buildWithNoMetadata() {
35          final String userMessage = "This is a message for you!";
36  
37          final Map<String, Object> output = new CreateUserRequestBuilder()
38              .setUserMessage(userMessage)
39              .build();
40  
41          boolean foundUserMessage = false;
42  
43          // We cannot call output.get() to inspect individual keys since build() produces a MultiMap
44          for (final Map.Entry<String, Object> entry : output.entrySet()) {
45              if (entry.getKey().equals(USER_MESSAGE_KEY)) {
46                  assertThat((String) entry.getValue(), is(userMessage));
47                  foundUserMessage = true;
48              } else {
49                  assertThat(entry.getValue(), is(nullValue()));
50              }
51          }
52  
53          assertTrue("User message was not in the output", foundUserMessage);
54      }
55  
56      @Test
57      public void buildWithMetadata() {
58          final String userMessage = "This is a message for you!";
59  
60          final Map<String, Object> metadata = new HashMap<>();
61          metadata.put("name", "fred");
62          metadata.put("age", 24);
63  
64          final Map<String, Object> output = new CreateUserRequestBuilder()
65              .setUserMessage(userMessage)
66              .setMetadata(metadata)
67              .build();
68  
69          boolean foundUserMessage = false;
70          boolean foundMetadata = false;
71  
72          // We cannot call output.get() to inspect individual keys since build() produces a MultiMap
73          for (final Map.Entry<String, Object> entry : output.entrySet()) {
74              final String key = entry.getKey();
75  
76              switch (key) {
77                  case USER_MESSAGE_KEY:
78                      assertThat(entry.getValue(), is(userMessage));
79                      foundUserMessage = true;
80                      break;
81                  case METADATA_KEY:
82                      //noinspection unchecked
83                      assertThat((List<Metadata<?>>) entry.getValue(), Matchers.<Object>containsInAnyOrder(
84                              new Metadata<>("name", "fred"),
85                              new Metadata<>("age", 24)
86                      ));
87  
88                      foundMetadata = true;
89                      break;
90                  default:
91                      assertThat(entry.getValue(), is(nullValue()));
92                      break;
93              }
94          }
95  
96          assertTrue("User message was not in the output", foundUserMessage);
97          assertTrue("Metadata was not in the output", foundMetadata);
98      }
