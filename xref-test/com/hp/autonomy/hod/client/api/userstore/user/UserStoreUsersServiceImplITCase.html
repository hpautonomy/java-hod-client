97      @Test
98      public void listUsersWithUserStoreUuid() throws HodErrorException {
99          final ResourceName userStoreName = getUserStore();
100         final ResourcesService resourcesService = new ResourcesServiceImpl(getConfig());
101 
102         final ListResourcesRequestBuilder listResourcesRequestBuilder = new ListResourcesRequestBuilder()
103                 .setDomains(Collections.singleton(userStoreName.getDomain()))
104                 .setTypes(Collections.singleton(ResourceType.USER_STORE));
105 
106         final ResourceUuid userStoreUuid = resourcesService.list(getTokenProxy(), listResourcesRequestBuilder).stream()
107             .filter(resourceDetails -> userStoreName.equals(resourceDetails.getResource().getResourceName()))
108             .findFirst()
109             .orElseThrow(() -> new IllegalStateException("Failed to retrieve user store UUID"))
110             .getResource().getResourceUuid();
111 
112         final List<User> uuidUsers = service.list(getTokenProxy(), userStoreUuid, false, false);
113         final List<User> nameUsers = service.list(getTokenProxy(), userStoreName, false, false);
114         assertThat(uuidUsers, hasSize(nameUsers.size()));
115     }

501     private UUID getUserUuidFromEmail(final String userEmail) throws HodErrorException {
502         final List<User> users = service.list(getTokenProxy(), getUserStore(), true, false);
503 
504         for (final User user : users) {
505             for (final Account account : user.getAccounts()) {
506                 if (Account.Type.EMAIL.equals(account.getType()) && userEmail.equals(account.getAccount())) {
507                     return user.getUuid();
508                 }
509             }
510         }
511 
512         return null;
513     }
