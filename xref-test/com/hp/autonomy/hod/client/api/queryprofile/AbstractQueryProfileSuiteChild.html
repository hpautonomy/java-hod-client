6   package com.hp.autonomy.hod.client.api.queryprofile;
7   
8   import com.hp.autonomy.hod.client.AbstractHodClientIntegrationTest;
9   import com.hp.autonomy.hod.client.Endpoint;
10  import com.hp.autonomy.hod.client.api.resource.ResourceName;
11  import com.hp.autonomy.hod.client.error.HodErrorException;
12  import lombok.Data;
13  import lombok.extern.slf4j.Slf4j;
14  import org.junit.After;
15  
16  import java.util.LinkedList;
17  import java.util.List;
18  import java.util.UUID;
19  
20  @Slf4j
21  public abstract class AbstractQueryProfileSuiteChild extends AbstractHodClientIntegrationTest {
22  
23      private List<ResourceName> createdQueryProfiles;
24  
25      // Not under test
26      private QueryProfileService queryProfileService;
27  
28      public AbstractQueryProfileSuiteChild(final Endpoint endpoint) {
29          super(endpoint);
30      }
31  
32      @Override
33      public void setUp() {
34          super.setUp();
35          queryProfileService = new QueryProfileServiceImpl(getConfig());
36          createdQueryProfiles = new LinkedList<>();
37      }
38  
39      @After
40      public void tearDown() {
41          for (final ResourceName profile : createdQueryProfiles) {
42              try {
43                  queryProfileService.deleteQueryProfile(getTokenProxy(), profile);
44              } catch (final HodErrorException e) {
45                  log.error("Error deleting query profile", e);
46              }
47          }
48      }
49  
50      // Safely create an empty query profile
51      protected ResponseAndIdentifier trackedCreateProfile() throws HodErrorException {
52          return trackedCreateProfile(new QueryProfileRequestBuilder());
53      }
54  
55      // Create a query profile and track it for deletion in tear down
56      protected ResponseAndIdentifier trackedCreateProfile(final QueryProfileRequestBuilder parameters) throws HodErrorException {
57          final String name = uniqueName();
58          final ResourceName profile = new ResourceName(getEndpoint().getDomainName(), name);
59          final QueryProfileStatusResponse response = queryProfileService.createQueryProfile(getTokenProxy(), name, QUERY_MANIPULATION_INDEX_NAME, parameters);
60          createdQueryProfiles.add(profile);
61          return new ResponseAndIdentifier(response, profile);
62      }
63  
64      // Delete a query profile and remove it from the list of tracked profiles
65      protected QueryProfileStatusResponse trackedDeleteProfile(final ResourceName profile) throws HodErrorException {
66          if (!createdQueryProfiles.contains(profile)) {
67              throw new IllegalStateException("Tried to delete profile which was not created with trackedCreateProfile: " + profile);
68          }
69  
70          final QueryProfileStatusResponse response = queryProfileService.deleteQueryProfile(getTokenProxy(), profile);
71          createdQueryProfiles.remove(profile);
72          return response;
73      }
74  
75      protected String uniqueName() {
76          return "java-client-" + UUID.randomUUID().toString();
77      }
78  
79      @Data
80      protected static class ResponseAndIdentifier {
81          private final QueryProfileStatusResponse response;
82          private final ResourceName profile;
83      }
84  }
