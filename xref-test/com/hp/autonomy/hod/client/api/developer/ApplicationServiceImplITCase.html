68          final String appName = randomName();
69          service.create(getDeveloperToken(), getEndpoint().getDomainName(), appName, APPLICATION_DESCRIPTION);

71          final List<Application> listPostCreate = service.list(getDeveloperToken(), getEndpoint().getDomainName());

212     private Optional<Application> applicationByName(final Collection<Application> applications, final String domain, final String name) {
213         return applications.stream()
214                 .filter(application -> domain.equals(application.getDomain()) && name.equals(application.getName()))
215                 .findFirst();
216     }
