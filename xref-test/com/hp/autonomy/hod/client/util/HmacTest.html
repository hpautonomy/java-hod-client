41      private void testHmacSign(final String expectedHmacToken, final String application) {
42          final String tokenId = "DF7aRd8VEeSiCdSFZKbA7w";
43          final String tokenSecret = "Ba90fFmxdioyouz06xr1fhn6Nxq4nB90jWEQ2UzDQr8";
44  
45          final AuthenticationToken<EntityType.Unbound, TokenType.HmacSha1> token = new AuthenticationToken<>(
46              EntityType.Unbound.INSTANCE,
47              TokenType.HmacSha1.INSTANCE,
48              new DateTime(123),
49              tokenId,
50              tokenSecret,
51              new DateTime(456)
52          );
53  
54          final Map<String, List<String>> queryParameters = new HashMap<>();
55          queryParameters.put("allowed_origins", Collections.singletonList("http://localhost:8080"));
56  
57          final Map<String, List<Object>> body = new HashMap<>();
58          body.put("domain", Collections.<Object>singletonList("IOD-TEST-DOMAIN"));
59          body.put("application", Collections.<Object>singletonList(application));
60          body.put("token_type", Collections.<Object>singletonList(TokenType.Simple.INSTANCE.getParameter()));
61  
62          final Request<String, Object> request = new Request<>(Request.Verb.POST, "/2/authenticate/combined", queryParameters, body);
63          final String hmacToken = hmac.generateToken(request, token);
64  
65          assertThat(hmacToken, is(expectedHmacToken));
66      }
